    \item Vidro de relógio,
    \item Lenço de papel,
    \item Espátula
\end{itemize}

\section{Reagentes}

Cloreto de Sódio. (NaCl)

\section{Procedimento Experimental}
Procedimento para pesagem por adição de 0,5000 g de NaCl com sensibilidade de 0,1 mg.

\begin{itemize}
    \item Verificar se a balança está nivelada.
    \item Ligar a balança e verificar se a mesma encontra-se “zerada”.
    \item Adicionar, cuidadosamente, 0,5000 g de NaCl, com auxílio de uma espátula, ao vidro de relógio.
    \item Fechar a porta da balança e efetuar a medida.
    \item Desligar a balança, retirar o vidro de relógio contendo o sal e verificar se a balança está limpa.
\end{itemize}
