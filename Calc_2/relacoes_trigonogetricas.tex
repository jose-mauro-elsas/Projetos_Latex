\section*{Identidades Trigonométricas (resumo)}
\addcontentsline{toc}{section}{Identidades Trigonométricas (resumo)}
\markright{Identidades Trigonométricas}

\small
\begin{description}\setlength{\itemsep}{0.8em}

\item[\textbf{Básicas (Pitágoras)}]
\[
\begin{aligned}
\sen^2 x + \cos^2 x &= 1\\
1+\tan^2 x &= \sec^2 x\\
1+\cot^2 x &= \csc^2 x
\end{aligned}
\]

\item[\textbf{Paridade}]
\[
\begin{aligned}
\sen(-x) &= -\sen x\\
\cos(-x) &= \cos x\\
\tan(-x) &= -\tan x
\end{aligned}
\]

\item[\textbf{Recíprocas e quociente}]
\[
\begin{aligned}
\csc x &= \frac{1}{\sen x}, \qquad
\sec x = \frac{1}{\cos x}, \qquad
\cot x = \frac{1}{\tan x}\\[2pt]
\tan x &= \frac{\sen x}{\cos x}, \qquad
\cot x = \frac{\cos x}{\sen x}
\end{aligned}
\]

\item[\textbf{Soma e diferença (seno/cosseno)} \quad \emph{(anote: “seno de (a+b)”)}]
\[
\begin{aligned}
\sen(a\pm b) &= \sen a\,\cos b \pm \cos a\,\sen b\\
\cos(a\pm b) &= \cos a\,\cos b \mp \sen a\,\sen b
\end{aligned}
\]

\item[\textbf{Soma e diferença (tangente)}]
\[
\begin{aligned}
\tan(a+b) &= \frac{\tan a+\tan b}{1-\tan a\,\tan b}\\
\tan(a-b) &= \frac{\tan a-\tan b}{1+\tan a\,\tan b}
\end{aligned}
\]

\item[\textbf{Ângulo duplo}]
\[
\begin{aligned}
\sen 2x &= 2\sen x\,\cos x\\
\cos 2x &= \cos^2 x-\sen^2 x
       = 1-2\sen^2 x
       = 2\cos^2 x-1\\
\tan 2x &= \frac{2\tan x}{1-\tan^2 x}
\end{aligned}
\]

\item[\textbf{Redução de potência (metade do ângulo)}]
\[
\begin{aligned}
\cos^2 x &= \frac12\,(1+\cos 2x)\\
\sen^2 x &= \frac12\,(1-\cos 2x)
\end{aligned}
\]

\item[\textbf{Meio-ângulo}]
\[
\begin{aligned}
\left|\sen\frac{x}{2}\right| &= \sqrt{\frac{1-\cos x}{2}}\\
\left|\cos\frac{x}{2}\right| &= \sqrt{\frac{1+\cos x}{2}}\\
\tan\frac{x}{2} &= \frac{1-\cos x}{\sen x}
               = \frac{\sen x}{1+\cos x}
\end{aligned}
\]

\item[\textbf{Produto $\rightarrow$ soma}]
\[
\begin{aligned}
\sen x\,\cos y &= \frac12\,[\sen(x-y)+\sen(x+y)]\\
\sen x\,\sen y &= \frac12\,[\cos(x-y)-\cos(x+y)]\\
\cos x\,\cos y &= \frac12\,[\cos(x-y)+\cos(x+y)]\\
\cos x\,\sen y &= \frac12\,[\sen(x+y)-\sen(x-y)]
\end{aligned}
\]

\item[\textbf{Diferença de senos (soma $\leftrightarrow$ produto)}]
\[
\sen x-\sen y=2\sen\!\left(\frac{x-y}{2}\right)\cos\!\left(\frac{x+y}{2}\right)
\]

\item[\textbf{Outras úteis}]
\[
\begin{aligned}
\sen x\,\cos x &= \frac12\,\sen 2x\\
1-\cos x &= 2\sen^2\!\left(\frac{x}{2}\right)\\
1+\cos x &= 2\cos^2\!\left(\frac{x}{2}\right)\\
1\pm \sen x &= 1\pm \cos\!\left(\frac{\pi}{2}-x\right)\\
\cos 2\theta &= \cos^2\theta-\sen^2\theta
\end{aligned}
\]

\end{description}
